def time_of_trip(datum, city):
    """

    elif city =='Washington':

    elif city =='Chicago':
        cday=datetime.strptime(datum['starttime'],'%m/%d/%Y %H:%M')

    # datetime.strptime(,'')
    # monthcday.monthhourcday.hour
    # cday.strftime('%A')str
    return (month, hour, day_of_week)

def number_of_trips(filename):

        #  csv reader 
        reader = csv.DictReader(f_in)
        
        # 
        n_subscribers = 0
        n_customers = 0
        
        # 
        for row in reader:
            if row['user_type'] == 'Subscriber':
                n_subscribers += 1
            else:
                n_customers += 1
        
        # 
        n_total = n_subscribers + n_customers
        #

        # 
        n_ct =n_customers/n_total

def duration_of_trips(filename):
    """
    
    
    """
    with open(filename, 'r') as f_in:
        #  csv reader 
        reader = csv.DictReader(f_in)
        
        # 
        n_long_times = 0
        n_normal_times = 0
        n_total_duration = 0
        
        # 
        for row in reader:
            n_total_duration += float(row['duration'])
            if float(row['duration']) > 30:
                n_long_times += 1
            else:
                n_normal_times += 1
        
        # 
        n_total_times = n_long_times + n_normal_times
        # 
        n_average_duration =n_total_duration/n_total_times
        # 
        n_la =n_long_times/n_total_times
        
        # 
        return(n_long_times, n_normal_times,n_total_times,n_total_duration,n_average_duration,n_la)

def month_distribution(filename):
    """
    
    
    """

        #  csv reader

def week_distribution(filename):
    """
    
    
    """
    with open(filename, 'r') as f_in:
        #  csv reader 
        reader = csv.DictReader(f_in)
        
        # 
        """
        
        week=[]
        for i in range(1,8):
            week.append(i)
        
        
        week=[i for i in range(1,8)]
        
        """
        # 
        week=[i for i in range(1,8)]
        dict_week_s=dict.fromkeys(week,0)
        dict_week_c=dict.fromkeys(week,0)

        # 
        for row in reader:
            if row['user_type']=='Subscriber':
                if row['day_of_week']=='Monday':
                    dict_week_s[1]+=1
                elif row['day_of_week']=='Tuesday':
                    dict_week_s[2]+=1
                elif row['day_of_week']=='Wednesday':
                    dict_week_s[3]+=1
                elif row['day_of_week']=='Thursday':
                    dict_week_s[4]+=1
                elif row['day_of_week']=='Friday':
                    dict_week_s[5]+=1
                elif row['day_of_week']=='Saturday':
                    dict_week_s[6]+=1
                elif row['day_of_week']=='Sunday':
                    dict_week_s[7]+=1
                
            elif row['user_type']=='Customer':
                if row['day_of_week']=='Monday':
                    dict_week_c[1]+=1
                elif row['day_of_week']=='Tuesday':
                    dict_week_c[2]+=1
                elif row['day_of_week']=='Wednesday':
                    dict_week_c[3]+=1
                elif row['day_of_week']=='Thursday':
                    dict_week_c[4]+=1
                elif row['day_of_week']=='Friday':
                    dict_week_c[5]+=1
                elif row['day_of_week']=='Saturday':
                    dict_week_c[6]+=1
                elif row['day_of_week']=='Sunday':
                    dict_week_c[7]+=1
        data_s=[]
        data_c=[]
        for i in range(1,8):
            data_s.append(dict_week_s[i])
            data_c.append(dict_week_c[i])
        return(data_s,data_c)

def hour_distribution(filename):
    """
    
    
    """
    with open(filename, 'r') as f_in:
        #  csv reader 
        reader = csv.DictReader(f_in)
        
        # 
        """
        
        hour=[]
        for i in range(24):
            hour.append(i)
        
        
        hour=[i for i in range(24)]
        
        """
        # 
        hour=[i for i in range(24)]
        dict_hour_s=dict.fromkeys(hour,0)
        dict_hour_c=dict.fromkeys(hour,0)
        # 
        for row in reader:
            if row['user_type']=='Subscriber':
                for i in range(24):
                    if int(row['hour'])==i:
                        dict_hour_s[i]+=1
                        

            elif row['user_type']=='Customer':
                for i in range(24):
                    if int(row['hour'])==i:
                        dict_hour_c[i]+=1
                        
        data_s=[]
        data_c=[]
        for i in range(24):
            data_s.append(dict_hour_s[i])
            data_c.append(dict_hour_c[i])
        return(data_s,data_c)
